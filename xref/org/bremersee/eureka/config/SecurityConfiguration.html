17  package org.bremersee.eureka.config;
18  
19  import org.bremersee.security.authentication.AuthenticationProperties;
20  import org.bremersee.security.authentication.PasswordFlowAuthenticationManager;
21  import org.bremersee.security.core.AuthorityConstants;
22  import org.springframework.beans.factory.ObjectProvider;
23  import org.springframework.boot.actuate.autoconfigure.security.servlet.EndpointRequest;
24  import org.springframework.boot.actuate.health.HealthEndpoint;
25  import org.springframework.boot.actuate.info.InfoEndpoint;
26  import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
27  import org.springframework.boot.context.properties.EnableConfigurationProperties;
28  import org.springframework.context.annotation.Bean;
29  import org.springframework.context.annotation.Configuration;
30  import org.springframework.core.annotation.Order;
31  import org.springframework.http.HttpMethod;
32  import org.springframework.security.config.annotation.web.builders.HttpSecurity;
33  import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
34  import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
35  import org.springframework.security.config.http.SessionCreationPolicy;
36  import org.springframework.security.core.userdetails.UserDetailsService;
37  import org.springframework.security.provisioning.InMemoryUserDetailsManager;
38  import org.springframework.security.web.util.matcher.AndRequestMatcher;
39  import org.springframework.security.web.util.matcher.AntPathRequestMatcher;
40  import org.springframework.security.web.util.matcher.NegatedRequestMatcher;
41  import org.springframework.util.Assert;
42  
43  /**
44   * The security configuration.
45   *
46   * @author Christian Bremer
47   */
48  @Configuration
49  @EnableWebSecurity
50  public class SecurityConfiguration {
51  
52    /**
53     * The only simple user authentication.
54     */
55    @ConditionalOnProperty(
56        prefix = "bremersee.security.authentication",
57        name = "enable-jwt-support",
58        havingValue = "false", matchIfMissing = true)
59    @Order(51)
60    @Configuration
61    @EnableConfigurationProperties(AuthenticationProperties.class)
62    static class OnlySimpleUserAuthentication extends WebSecurityConfigurerAdapter {
63  
64      private AuthenticationProperties properties;
65  
66      /**
67       * Instantiates a only simple user authentication.
68       *
69       * @param properties the properties
70       */
71      public OnlySimpleUserAuthentication(AuthenticationProperties properties) {
72        this.properties = properties;
73      }
74  
75      @Override
76      protected void configure(HttpSecurity http) throws Exception {
77        http
78            .authorizeRequests()
79            .antMatchers(HttpMethod.OPTIONS).permitAll()
80            .antMatchers("/eureka/css/**", "/eureka/js/**", "/eureka/fonts/**",
81                "/eureka/images/**").permitAll()
82            .requestMatchers(EndpointRequest.to(HealthEndpoint.class)).permitAll()
83            .requestMatchers(EndpointRequest.to(InfoEndpoint.class)).permitAll()
84            .requestMatchers(new AndRequestMatcher(
85                EndpointRequest.toAnyEndpoint(),
86                new AntPathRequestMatcher("/**", HttpMethod.GET.name())))
87            .access(properties.getActuator().buildAccessExpression(properties::ensureRolePrefix))
88            .requestMatchers(EndpointRequest.toAnyEndpoint())
89            .access(properties.getActuator().buildAdminAccessExpression(properties::ensureRolePrefix))
90            .antMatchers("/eureka/**")
91            .access(properties.getEureka().buildAccessExpression(properties::ensureRolePrefix))
92            .anyRequest()
93            .access(properties.getApplication().buildAccessExpression(
94                false, true, false, true,
95                properties::ensureRolePrefix,
96                AuthorityConstants.ADMIN_ROLE_NAME))
97            .and()
98            .userDetailsService(userDetailsService())
99            .csrf().disable()
100           .formLogin().disable()
101           .httpBasic().realmName("eureka");
102     }
103 
104     @Bean
105     @Override
106     public UserDetailsService userDetailsService() {
107       return new InMemoryUserDetailsManager(properties.getEureka().buildBasicAuthUserDetails(
108           properties.buildBasicAuthUserDetails()));
109     }
110   }
111 
112   /**
113    * The eureka authentication.
114    */
115   @ConditionalOnProperty(
116       prefix = "bremersee.security.authentication",
117       name = "enable-jwt-support",
118       havingValue = "true")
119   @Order(51)
120   @Configuration
121   @EnableConfigurationProperties(AuthenticationProperties.class)
122   static class EurekaAuthentication extends WebSecurityConfigurerAdapter {
123 
124     private AuthenticationProperties properties;
125 
126     /**
127      * Instantiates a new eureka authentication.
128      *
129      * @param properties the properties
130      */
131     public EurekaAuthentication(AuthenticationProperties properties) {
132       this.properties = properties;
133     }
134 
135     @Override
136     protected void configure(HttpSecurity http) throws Exception {
137       http
138           .requestMatcher(new AntPathRequestMatcher("/eureka/**"))
139           .authorizeRequests()
140           .antMatchers(HttpMethod.OPTIONS).permitAll()
141           .antMatchers("/eureka/css/**", "/eureka/js/**", "/eureka/fonts/**",
142               "/eureka/images/**").permitAll()
143           .anyRequest()
144           .access(properties.getEureka().buildAccessExpression(properties::ensureRolePrefix))
145           .and()
146           .userDetailsService(userDetailsService())
147           .csrf().disable()
148           .formLogin().disable()
149           .httpBasic().realmName("eureka");
150     }
151 
152     @Bean
153     @Override
154     public UserDetailsService userDetailsService() {
155       return new InMemoryUserDetailsManager(properties.getEureka().buildBasicAuthUserDetails());
156     }
157   }
158 
159   /**
160    * The password flow authentication.
161    */
162   @ConditionalOnProperty(
163       prefix = "bremersee.security.authentication",
164       name = "enable-jwt-support",
165       havingValue = "true")
166   @Order(52)
167   @Configuration
168   @EnableConfigurationProperties(AuthenticationProperties.class)
169   static class PasswordFlowAuthentication extends WebSecurityConfigurerAdapter {
170 
171     private final AuthenticationProperties properties;
172 
173     private final PasswordFlowAuthenticationManager passwordFlowAuthenticationManager;
174 
175     /**
176      * Instantiates a new password flow authentication.
177      *
178      * @param properties the properties
179      * @param passwordFlowAuthenticationManager the password flow authentication manager
180      */
181     public PasswordFlowAuthentication(
182         AuthenticationProperties properties,
183         ObjectProvider<PasswordFlowAuthenticationManager> passwordFlowAuthenticationManager) {
184       this.properties = properties;
185       this.passwordFlowAuthenticationManager = passwordFlowAuthenticationManager.getIfAvailable();
186       Assert.notNull(
187           this.passwordFlowAuthenticationManager,
188           "Password flow authentication manager must be present.");
189     }
190 
191     @Override
192     protected void configure(HttpSecurity http) throws Exception {
193       http
194           .requestMatcher(new NegatedRequestMatcher(new AntPathRequestMatcher("/eureka/**")))
195           .authorizeRequests()
196           .antMatchers(HttpMethod.OPTIONS).permitAll()
197           .requestMatchers(EndpointRequest.to(HealthEndpoint.class)).permitAll()
198           .requestMatchers(EndpointRequest.to(InfoEndpoint.class)).permitAll()
199           .requestMatchers(new AndRequestMatcher(
200               EndpointRequest.toAnyEndpoint(),
201               new AntPathRequestMatcher("/**", HttpMethod.GET.name())))
202           .access(properties.getActuator().buildAccessExpression(properties::ensureRolePrefix))
203           .requestMatchers(EndpointRequest.toAnyEndpoint())
204           .access(properties.getActuator().buildAdminAccessExpression(properties::ensureRolePrefix))
205           .anyRequest()
206           .access(properties.getApplication().buildAccessExpression(
207               false, true, false, true,
208               properties::ensureRolePrefix,
209               AuthorityConstants.ADMIN_ROLE_NAME))
210           .and()
211           .sessionManagement()
212           .sessionCreationPolicy(SessionCreationPolicy.STATELESS)
213           .and()
214           .authenticationProvider(passwordFlowAuthenticationManager)
215           .csrf().disable()
216           .formLogin().disable()
217           .httpBasic()
218           .realmName("actuator");
219     }
220   }
